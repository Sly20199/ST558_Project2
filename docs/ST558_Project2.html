Data Cleaning

In this step, we clean and preprocess the raw dataset to ensure all variables are in the correct format for statistical modeling.

First, we use ‘na.if()’ across all character columns to replace the empty string entries with formal ‘NA’ missing values.

Next, we convert categorical variables into R ‘factor’ objects and explicitly specify their levels.

Finally, we convert game score, time remaining, and shot distance variables.

NBA_data <- NBA_data %>%
  mutate(
    # Convert empty strings to NA
    across(where(is.character), ~ na_if(., "")),
    
    # Factor levels
    GameType = factor(GameType, levels = c("playoff", "regular")),
    WinningTeam = factor(WinningTeam, levels = c(
      "TOR", "LAC", "CHO", "DET", "ORL", "MIN", "MIA", "PHI", "DAL", "SAS",
      "UTA", "PHO", "DEN", "ATL","MIL", "BOS", "BRK", "CHI", "WAS", "POR",
      "LAL", "CLE", "HOU", "OKC", "MEM", "NYK", "GSW", "IND", "NOP","SAC"
      )),
   Quarter = factor(Quarter, levels = c(1, 2, 3, 4, 5, 6)),
   AwayTeam = factor(AwayTeam, levels = c(
     "NOP", "LAL", "CHI", "DET", "CLE", "MIN","MEM", "BOS", "WAS", "NYK", "OKC", "SAC",
     "DEN", "ATL", "MIL", "LAC", "TOR", "DAL","PHO", "POR", "UTA", "MIA", "PHI", "ORL",
     "IND", "GSW", "BRK", "CHO", "HOU", "SAS"
     )),
   HomeTeam = factor(HomeTeam, levels = c(
     "TOR", "LAC", "CHO", "IND", "ORL", "BRK", "MIA", "PHI", "DAL", "SAS", "UTA",
     "PHO", "POR", "DET", "HOU", "GSW", "BOS", "MEM", "NOP", "OKC", "DEN", "SAC",
     "LAL", "MIL", "ATL", "NYK", "CHI", "CLE", "MIN", "WAS" 
     )),
   ShotType = factor(ShotType, 
                          levels = c( "2-pt jump shot",
                                      "2-pt layup",
                                      "3-pt jump shot",
                                      "2-pt hook shot",
                                      "2-pt dunk")),
   ShotOutcome = factor(ShotOutcome, levels = c("miss","make")),
   FoulType = factor(FoulType,levels = c(
     "shooting", "offensive", "personal", "loose ball","technical", 
     "away from play","personal take", "flagrant", "clear path"
     )),
   FreeThrowOutcome = factor(FreeThrowOutcome, levels = c("make", "miss")),
   FreeThrowNum = factor(FreeThrowNum, levels = c(
    "1 of 2", "2 of 2", "1 of 1", "1 of 3", "2 of 3", "3 of 3", "technical"
     )),
   ViolationType = factor(ViolationType, levels = c(
     "kicked ball", "def goaltending", "delay of game", "jump ball", "double lane",
      "violation"
     )),
  TimeoutTeam = factor(TimeoutTeam, levels = c(
    "TOR", "NOP", "LAC", "LAL", "CHO", "CHI", "IND", "DET", "ORL", "CLE",
    "BRK", "MIN", "MEM", "MIA", "PHI", "BOS", "DAL", "WAS", "NYK", "SAS", "UTA",
    "OKC", "PHO", "SAC", "POR", "DEN", "ATL", "MIL", "HOU", "GSW"
    )),
  TurnoverType = factor(TurnoverType, levels = c(
      "offensive foul", "bad pass", "traveling", "out of bounds lost ball", "8 sec",
      "lost ball", "step out of bounds", "back court", "3 sec", "shot clock", "palming",
      "off goaltending", "dbl dribble", "illegal assist", "turnover", "lane violation",
      "discontinued dribble", "inbound", "5 sec"
      )),

  ReboundType = factor( ReboundType, levels = c("offensive", "defensive")),

Numerical summaries

Filter Missing Values and Summarize Shot Distance by Outcome

Before computing summary statistics, we filter the dataset to create ‘NBA_data_clean’, removing any rows with missing values (‘NA’) in our primary analysis variables: ‘ShotOutcome’, ‘ShotDist’, and ‘GameType’.

We then calculate numerical summaries for the ‘ShotDist’(Shot Distance) variable, grouped by the ‘ShotOutcome’(make vs. miss). The summary metrics include the sample size, mean, median, standard deviation, minimum, and maximum. This helps us examine whether the average distance of a shot differs between successful and unsuccessful attempts.

# Create clean data set
NBA_data_clean <- NBA_data  %>%
  filter(!is.na(ShotOutcome),
         !is.na(ShotDist),
         !is.na(GameType)) 

# Numerical Summary
NBA_data_clean %>%
  group_by(ShotOutcome)  %>%
  summarise(
    n = n(),
    mean = mean(ShotDist, na.rm = TRUE),
    median = median(ShotDist, na.rm = TRUE),
    sd = sd(ShotDist, na.rm = TRUE),
    min = min(ShotDist, na.rm = TRUE),
    max = max(ShotDist, na.rm = TRUE)
  )

Plot 3: Side-by-side Bar Chart (Multivariate - Shot Outcomes by Game Type)

To investigate whether players’ shot outcomes differ between high-pressure playoff games compared to regular season games, we create a side-by-side bar chart. We use fill = GameType to color the bars by game type and set position = “dodge” to place the playoff and regular season bars next to each other for easy comparison.

ggplot(NBA_data_clean, aes(x = ShotOutcome, fill = GameType)) +
  geom_bar(position = "dodge", alpha = 0.9) +
  scale_fill_manual(values = c("playoff" = "#ff6e40", "regular" = "#1e3d59")) +
  labs(
    title = "Counts of Shot Outcomes by Game Type",
    x = "Shot Outcome",
    y = "Count",
    fill = "Game Type"
  )

Plot 4: Box Plot (Multivariate - Shot Distance by Shot Outcomes)

To analyze the spread and key statistics of shot distances grouped by shot success, we construct side-by-side box plots. This visualization helps us assess the relationship between how far a shot is taken and whether it is successfully made.

ggplot(NBA_data_clean , aes(x = ShotOutcome, y = ShotDist, fill = ShotOutcome)) +

Plot 5: Faceted Histogram (Multivariate & Faceted - Shot Distance by Quarter)

To analyze how shot distance and success rates evolve as the game progresses, we construct a faceted histogram.

ggplot(NBA_data_clean, aes(x = ShotDist, fill = ShotOutcome)) +

Plot 7: Bivariate Highest Density Region (HDR) Plot

To investigate whether shot distances differ significantly among the top three most frequent shot types, we construct a categorical statistical comparison plot using ggbetweenstats().

top3_shot_types <- NBA_data_clean %>% 
  dplyr::count(ShotType, sort = TRUE) %>%
  slice_head(n = 3) %>%
  pull(ShotType)

NBA_top3 <- NBA_data_clean %>% 
  filter(ShotType %in% top3_shot_types)

if (nrow(NBA_top3) > 300){
  set.seed(123)
  NBA_top3 <- NBA_top3[sample(1:nrow(NBA_top3), 300), ]
}

ggbetweenstats(
  data  = NBA_top3,
  x     = ShotType, 
  y     = ShotDist,
  title = "Shot Distance by Shot Type(Top3)",
  xlab  = "Shot Type",
  ylab  = "Shot Distance(feet)",
  type  = "nonparametric",
  pairwise.comparisons = TRUE, 
  p.adjust.method = "BH"
)
